# Import module
>>> MyList = __import__('1-my_list').MyList

# Test class
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]

# Test issubclass
>>> issubclass(MyList, list)
True

# Test isinstance
>>> my_list = MyList()
>>> isinstance(my_list, list)
True

# Test initial print_sorted method
>>> my_list.print_sorted()
[]  # Since the list is initially empty

# Test print_sorted method
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]

# Test again to check if the original list is unmodified
>>> print(my_list)
[1, 4, 2, 3, 5]

# Add more elements to test sorting with new elements
>>> my_list.append(5)
>>> my_list.append(1)
>>> my_list.append(8)
>>> my_list.append(0)
>>> my_list.append(3)

# Print the list to check if the new elements are added
>>> print(my_list)
[1, 4, 2, 3, 5, 5, 1, 8, 0, 3]

# Print the sorted list to check if it is sorting correctly
>>> my_list.print_sorted()
[0, 1, 1, 2, 3, 3, 4, 5, 5, 8]

# Check if the original list is still unmodified after sorting
>>> print(my_list)
[1, 4, 2, 3, 5, 5, 1, 8, 0, 3]
